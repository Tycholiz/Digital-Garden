    return a > b ? 1 : -1;
})

Sort by multiple properties

list.sort((a, b) => {
    if (b.ratings !== a.ratings) {
        // first sort by ratings (desc)
        return b.ratings - a.ratings
    } else {
        // then sort by id (desc)
        return b.id - a.id
    }
})
